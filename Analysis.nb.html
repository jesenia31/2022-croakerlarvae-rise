#Question set to answer first main question relating morphological traits within the years/location

Hypothesis: Fish from 1995 will have a higher total length and aspect ratio, and a lower square caudal peduncle depth factor compared to fish from 2017.

#Make dataframe for 1995 & 2017 fish
##in order to create statistical tests
atcroaker95 <- atlantic_croaker_final %>% 
  filter(Year==1995)

atcroaker17 <- atlantic_croaker_final %>% 
  filter(Year==2017)

Question 1: Is aspect ratio higher in fish from 1995 than fish from 2017?

#Making a plot
